> 模拟，进程

###题意
分组比赛，每组胜者进下轮，直到角出冠军

###细节
- 本轮被淘汰者的排名都为下轮人数+1

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;
int r[1010],a[1010],p[2][1010];
int main(){
    freopen("d.in","r",stdin);
    int n,m,i,j,t,mm,temp,f=0,cnt,N;
    scanf("%d%d",&n,&m);
    N=n;
    for(i=0;i<n;++i) scanf("%d",&a[i]);
    for(i=0;i<n;++i) scanf("%d",&p[0][i]);
    while(n>1){
        //for(i=0;i<n;++i) printf("%d; ",p[f][i]); printf("\n");
        t=(n-1)/m+1;
        cnt=0;
        for(i=0;i<n;i+=m){
            mm=-1;
            for(j=i;j<i+m && j<n;++j){
                if(a[p[f][j]]>mm){
                    mm=a[p[f][j]];
                    temp=p[f][j];
                }
            }
            for(j=i;j<i+m && j<n;++j){
                if(a[p[f][j]]!=mm){
                    r[p[f][j]]=t;
                }
            }
            p[1-f][cnt++]=temp;
        }
        f=1-f;
        n=t;
    }
    printf("%d",r[0]+1);
    for(i=1;i<N;++i) printf(" %d",r[i]+1);
    return 0;
}
```
