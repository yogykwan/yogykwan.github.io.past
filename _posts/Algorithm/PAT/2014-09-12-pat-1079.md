> 树形DP

###题意
给出供求关系，求所有零售商销售总额

###细节
- 递归至叶子结点直接结算

### 代码
```
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#include<queue>
using namespace std;

struct E{
    int v,nxt;
}e[100010];
int head[100010],cnt;
double sale[100010],d[100010],r,price,ans;
void add(int u,int v){
    e[cnt].v=v;
    e[cnt].nxt=head[u];
    head[u]=cnt++;
}
void gao(int u){
    if(head[u]==-1){
        ans+=d[u]*sale[u];
        return;
    }
    int i,v;
    for(i=head[u];i!=-1;i=e[i].nxt){
        v=e[i].v;
        d[v]=d[u]*r;
        gao(v);
    }
}
int main(){
    freopen("d.in","r",stdin);
    int n,m,i,j;
    scanf("%d%lf%lf",&n,&price,&r);
    r=1+r/100;
    memset(head,-1,sizeof(head));
    for(i=0;i<n;++i){
        scanf("%d",&m);
        if(m==0){
            scanf("%lf",&sale[i]);
        }else{
            while(m--){
                scanf("%d",&j);
                add(i,j);
            }
        }
    }
    d[0]=1;
    gao(0);
    printf("%.1f\n",ans*price);
    return 0;
}

```
