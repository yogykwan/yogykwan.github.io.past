> 模拟

###题意
求两点最短距离

###细节
- 记录正向距离做差值

### 代码
```
#include<cstdio>
#include<cstring>

int d[100010],s[100010];

int main(){
    freopen("d.in","r",stdin);
    int n,i,m,sum=0,u,v,ans;
    scanf("%d",&n);
    for(i=1;i<=n;++i){
        scanf("%d",&d[i]);
        s[i]=s[i-1]+d[i];
        sum+=d[i];
    }
    scanf("%d",&m);
    while(m--){
        scanf("%d%d",&u,&v);
        if(u==v){
            printf("0\n");
            continue;
        }
        if(u>v){
            i=u;u=v;v=i;
        }
        ans=s[v-1]-s[u-1];
        if(sum-ans<ans) ans=sum-ans;
        printf("%d\n",ans);
    }
    return 0;
}

```
