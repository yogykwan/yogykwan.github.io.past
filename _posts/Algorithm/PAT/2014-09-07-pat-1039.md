> 模拟，邻接表

###题意
求每人的选修科目

###细节
- <string>超时，使用哈希处理字符串姓名
- <set>超内存，使用邻接表
- 最多输入500000个姓名

### 代码
```
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
#include<iostream>
#include<set>
#include<iterator>
#include<map>
#include<queue>
using namespace std;

struct E{
    int next;
    short int v;
}e[500000];
int head[175760];
char st[5];
short int ans[2510];
int tot;
int cnt;
void add(int u,short int v){
    e[tot].next=head[u];
    e[tot].v=v;
    head[u]=tot++;
}
int main(){
    freopen("d.in","r",stdin);
    int n,m,i,j,k;
    short int x;
    memset(head,-1,sizeof(head));
    scanf("%d%d",&m,&n);
    while(n--){
        scanf("%hd%d",&x,&i);
        for(j=0;j<i;++j){
            scanf("%s",st);
            k=st[3]-'0'+10*(st[2]-'A')+260*(st[1]-'A')+6760*(st[0]-'A');
            add(k,x);
        }
    }
    while(m--){
        scanf("%s",st);
        k=st[3]-'0'+10*(st[2]-'A')+260*(st[1]-'A')+6760*(st[0]-'A');
        cnt=0;
        for(i=head[k];i!=-1;i=e[i].next){
            ans[cnt++]=e[i].v;
        }
        printf("%s %d",st,cnt);
        sort(ans,ans+cnt);
        for(i=0;i<cnt;++i) printf(" %hd",ans[i]);
        printf("\n");
    }
    return 0;
}
```
