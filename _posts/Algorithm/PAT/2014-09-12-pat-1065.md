> 模拟，大数

###题意
判断两数和是否大于另一数

###细节
- 输入包含的2^63已超long long
- 单独记录符号和绝对值，后者unsigned long long
- 两个2^63相加时特判

### 代码
```
#include<cstdio>
#include<iostream>
using namespace std;

typedef unsigned long long ull;
ull a,b,c;
int fa,fb,fc;

void gaoin(ull &a,int &f){
    char ch;
    cin>>ch;
    if(ch=='-') f=1;
    else cin.putback(ch);
    cin>>a;
}
bool gao(){
    if(!fa && fc) return 1;
    if(fa && !fc) return 0;
    if(!fa && !fc) return a>c;
    return a<c;
}
int main(void) {
    freopen("d.in","r",stdin);
    int n,i;
    ull m=1;
    m=m<<63;
    cin>>n;
    getchar();
    for(i=1;i<=n;++i){
        fa=fb=fc=0;
        gaoin(a,fa);
        gaoin(b,fb);
        gaoin(c,fc);
        printf("Case #%d: ",i);
        if(!fa && !fb){
            if(fc || a==m && b==m) printf("true\n");
            else{
                if(a+b>c) printf("true\n");
                else printf("false\n");
            }
        }else if(fa && fb){
            if(!fc || a==m && b==m) printf("false\n");
            else{
                if(a+b<c) printf("true\n");
                else printf("false\n");
            }
        }else if(!fa && fb){
            if(a>=b){
                fa=0;
                a-=b;
            }else{
                fa=1;
                a=b-a;
            }
            printf("%s\n",gao()?"true":"false");
        }else{
            if(a>b){
                fa=1;
                a-=b;
            }else{
                fa=0;
                a=b-a;
            }
            printf("%s\n",gao()?"true":"false");
        }
    }
    return 0;
}
```
