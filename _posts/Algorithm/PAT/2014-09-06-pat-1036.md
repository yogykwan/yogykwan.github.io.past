> 模拟，遍历

###题意
找出最低男和最高女，求差值

###细节
- 直接分男女比较

### 代码
```
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
#include<iostream>
#include<map>
using namespace std;
char name[2][20],id[2][20];
char a[20],b[5],c[20];
int grade[2];
int d;
void gao(int i){
    if(grade[i]==-1 || i==0 && d>grade[i] || i==1 && d<grade[i]){
        strcpy(name[i],a);
        strcpy(id[i],c);
        grade[i]=d;
    }
}
int main(){
    freopen("d.in","r",stdin);
    int n,i;
    grade[0]=grade[1]=-1;
    scanf("%d",&n);
    for(i=0;i<n;++i){
        scanf("%s%s%s%d",a,b,c,&d);
        if(b[0]=='F') gao(0);
        else gao(1);
    }
    for(i=0;i<2;++i){
        if(grade[i]!=-1) printf("%s %s\n",name[i],id[i]);
        else printf("Absent\n");
    }
    if(grade[0]!=-1 && grade[1]!=-1) printf("%d\n",grade[0]-grade[1]);
    else printf("NA\n");
    return 0;
}


```
