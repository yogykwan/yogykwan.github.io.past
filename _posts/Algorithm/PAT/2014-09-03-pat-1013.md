> 图论

###题意
求某城市被攻占后，使全局连通的修路数目

###细节
- 每次求新的最小生成树

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
struct E{
    int u,v;
}e[1000010];
int fa[1010];
int ff(int x){
    if(fa[x]!=x) fa[x]=ff(fa[x]);
    return fa[x];
}
int main(){
    freopen("d.in","r",stdin);
    int n,m,k,cnt,i,j,u,v,x;
    scanf("%d%d%d",&n,&m,&k);
    for(i=0;i<m;++i) scanf("%d%d",&e[i].u,&e[i].v);
    while(k--){
        scanf("%d",&x);
        cnt=0;
        for(i=1;i<=n;++i) fa[i]=i;
        for(j=0;j<m;++j){
            if(e[j].u==x || e[j].v==x) continue;
            u=ff(e[j].u);
            v=ff(e[j].v);
            if(u!=v){
                fa[u]=v;
                ++cnt;
                if(cnt==n-2) break;
            }
        }
        printf("%d\n",n-2-cnt);
    }
    return 0;
}


```
