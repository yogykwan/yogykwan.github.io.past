> 二分

###题意
求和为特定值的子序列

###细节
- 不存在则输出大于要求的最小序列
- 二分注意上下界

### 代码
```
#include<cstdio>
#include<cstring>

int pre[100010],s[100010],d[100010],a[100010];

int main(){
    freopen("d.in","r",stdin);
    int n,D,l,r,i,m;
    scanf("%d%d",&n,&D);
    for(i=1;i<=n;++i){
        scanf("%d",&a[i]);
        s[i]=s[i-1]+a[i];
    }
    for(i=1;i<=n;++i){
        if(s[i]<D) continue;
        if(s[i]==D){
            pre[i]=1;
            d[i]=D;
            continue;
        }
        l=1,r=i;
        while(l<r){
            m=(l+r)>>1;
            if(s[i]-s[m-1]==D){
                pre[i]=m;
                d[i]=D;
                goto L;
            }
            if(s[i]-s[m-1]<D) r=m-1;
            else l=m+1;
        }
        if(s[i]-s[r-1]>=D){
            pre[i]=r;
            d[i]=s[i]-s[r-1];
            continue;
        }
        l=r-1;
        pre[i]=l;
        d[i]=s[i]-s[l-1];
        L:;
    }
    int mm=100000001;
    for(i=1;i<=n;++i)
        if(pre[i] && d[i]<mm)
            mm=d[i];
    for(i=1;i<=n;++i)
        if(d[i]==mm) 
            printf("%d-%d\n",pre[i],i);
    return 0;
}
```
