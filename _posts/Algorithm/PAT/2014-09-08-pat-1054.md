> 模拟，统计

###题意
找到出现最多次的值

###细节
- 排序后线性统计

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int a[480010];
int main(){
    freopen("d.in","r",stdin);
    int i,j,n,ans,m,cnt;
    scanf("%d%d",&i,&j);
    n=i*j;
    for(i=0;i<n;++i) scanf("%d",a+i);
    sort(a,a+n);
    a[n]=-1;
    m=1;
    cnt=0;
    for(i=1;i<=n;++i){
        if(a[i]==a[i-1]) ++m;
        else{
            if(m>cnt){
                cnt=m;
                ans=a[i-1];
            }
            m=1;
        }
    }
    printf("%d",ans);
    return 0;
}

```
