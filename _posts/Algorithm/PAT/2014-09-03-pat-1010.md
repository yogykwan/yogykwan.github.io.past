> 二分

###题意
求使数值相等的进制

###细节
- 进制转换用long long
- 二分下届为待测串中最大字符值、上届为已知串的数值

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

typedef long long ll;
const ll inf=0x7fffffffffffffff;
char a[15],b[15],c[15];
ll da,db;

ll gao(char *a,ll r){
    ll ans=0;
    for(int i=0;a[i];++i){
        if(a[i]<='9'){
            ans=ans*r+a[i]-48;
        }else{
            ans=ans*r+a[i]-87;
        }
        if(ans<0 || ans>da) return inf;
    }
    return ans;
}

int cnta,cntb,n;
int main(){
    int i,type;
    ll l,r,m;
    freopen("d.in","r",stdin);
    scanf("%s%s%d%lld",a,b,&type,&r);
    if(type==2){
        strcpy(c,a);
        strcpy(a,b);
        strcpy(b,c);
    }
    da=inf;
    da=gao(a,r);
    r=0;
    for(i=0;b[i];++i) if(b[i]>r) r=b[i];
    if(r<='9') r-=47;
    else r-=86;
    l=r;r=da;
    while(l<r){
        m=(l+r)>>1;
        db=gao(b,m);
        if(db<da) l=m+1;
        else if(db>da) r=m-1;
        else r=m;
    }
    if(gao(b,l)==da) printf("%lld\n",l);
    else if(gao(b,r)==da) printf("%lld\n",r);
    else printf("Impossible\n");
    return 0;
}
```
