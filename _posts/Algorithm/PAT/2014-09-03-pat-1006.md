> 排序

###题意
求最早来和最晚走的人

###细节
- 时间字符串可直接比较

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

char open[20],close[20];
char id[20],in[20],out[20];
char a[20],b[20];
int main(){
    freopen("d.in","r",stdin);
    int n,i;
    scanf("%d",&n);
    a[0]='9';
    b[0]='/';
    for(i=0;i<n;++i){
        scanf("%s%s%s",id,in,out);
        if(strcmp(in,a)<0){
            strcpy(a,in);
            strcpy(open,id);
        }
        if(strcmp(out,b)>0){
            strcpy(b,out);
            strcpy(close,id);
        }
    }

    printf("%s %s\n",open,close);
    return 0;
}


```
