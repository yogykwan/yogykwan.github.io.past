###题意
根据置换序列多次洗牌

###细节
- 直接遍历

### 代码
```
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
#include<iostream>
#include<set>
#include<iterator>
#include<map>
#include<queue>
using namespace std;
char f[5]={'S','H','C','D','J'};
int s[60],a[22][60];
void gao(int x){
    printf("%c",f[(x-1)/13]);
    printf("%d",(x-1)%13+1);
}
int main(){
    freopen("d.in","r",stdin);
    int n,i,j;
    scanf("%d",&n);
    for(i=1;i<=54;++i) scanf("%d",&s[i]),a[0][i]=i;
    for(j=1;j<=n;++j)
        for(i=1;i<=54;++i)
            a[j][s[i]]=a[j-1][i];
    for(i=1;i<54;++i){
        gao(a[n][i]);
        printf(" ");
    }
    gao(a[n][i]);
    return 0;
}

```
