> 模拟，大数

###题意
求某数某进制表达下的正反序是否都为素数

###细节
- 输入为十进制

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<queue>
#include<iostream>
#include<cmath>
using namespace std;
int b[100];
int is(int a){
    if(a<2) return 0;
    for(int i=2;i*i<=a;++i) if(a%i==0) return 0;
    return 1;
}
int main(){
    freopen("d.in","r",stdin);
    int n,d,i,x,y,m;
    while(scanf("%d",&n)!=EOF && n>=0){
        scanf("%d",&d);
        int m=0;
        x=n;
        while(n){
            b[m++]=n%d;
            n/=d;
        }
        y=0;
        for(i=0;i<m;++i){
            y=y*d+b[i];
        }
        if(is(x) && is(y)) printf("Yes\n");
        else printf("No\n");
    }
    return 0;
}


```
