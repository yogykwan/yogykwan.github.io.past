> DFS，BFS

###题意
给出二叉树的后序遍历、中序遍历，求分层遍历

###细节
- DSF重构二叉树
- BFS输出分层遍历

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<queue>
#include<algorithm>
#include<cmath>
#include<string>
using namespace std;
int n,p;
int post[50],in[50];
int l[50],r[50];
int ans[50];
int gao(int ll,int rr){
    if(ll>rr) return -1;
    int i;
    for(i=ll;i<=rr;++i) if(in[i]==post[p]) break;
    --p;
    r[i]=gao(i+1,rr);
    l[i]=gao(ll,i-1);
    return i;
}
int main(){
    freopen("d.in","r",stdin);
    scanf("%d",&n);
    int i,rt,fr,rear;
    for(i=0;i<n;++i) scanf("%d",&post[i]);
    for(i=0;i<n;++i) scanf("%d",&in[i]);
    p=n-1;
    ans[0]=gao(0,n-1);
    fr=0,rear=1;
    while(rear<n){
        rt=ans[fr++];
        if(l[rt]!=-1) ans[rear++]=l[rt];
        if(r[rt]!=-1) ans[rear++]=r[rt];
    }
    for(i=0;i<n-1;++i) printf("%d ",in[ans[i]]);
    printf("%d\n",in[ans[i]]);
    return 0;
}


```
