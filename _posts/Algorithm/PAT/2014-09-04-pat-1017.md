> 数据结构

###题意
求平均等待时间

###细节
- 业务完成时间放入优先队列
- 17：00前到达则必须办理

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<queue>
#include<algorithm>
#include<cmath>
#include<string>
using namespace std;
struct P{
    int a,l;
}p[10010];
bool cmp(P a,P b){
    return a.a<b.a;
}
priority_queue<int,vector<int>,greater<int> > pq;

int main(){
    freopen("d.in","r",stdin);
    int n,m,i,j,hh,mm,ss,now=3600*8,ans=0;
    scanf("%d%d",&n,&m);
    for(i=0;i<n;++i){
        scanf("%d:%d:%d%d",&hh,&mm,&ss,&p[i].l);
        p[i].l*=60;
        if(hh>=17){
            --i;--n;
            continue;
        }
        p[i].a=hh*3600+mm*60+ss;
    }
    sort(p,p+n,cmp);
    for(i=j=0;i<n && j<m;++i,++j){
        if(p[i].a>now) now=p[i].a;
        pq.push(now+p[i].l);
        ans+=now-p[i].a;
    }
    for(;i<n;++i){
        now=pq.top();
        pq.pop();
        if(p[i].a>now) now=p[i].a;
        pq.push(now+p[i].l);
        ans+=now-p[i].a;
    }
    printf("%.1f\n",ans*0.1/6/n);
    return 0;
}


```
