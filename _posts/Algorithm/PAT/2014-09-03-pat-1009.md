> 模拟

###题意
多项式求积

###细节
- 系数“0.0”或“-0.0”不输出
- 积的指数范围为输入多项式两倍

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
int ae[15],be[15];
double ac[15],bc[15];
double cc[2222];

int cnta,cntb,n;
int main(){
    freopen("d.in","r",stdin);
    int i,j;
    for(i=0;i<=2000;++i) cc[i]=0;
    scanf("%d",&cnta);
    for(i=0;i<cnta;++i){
        scanf("%d%lf",&ae[i],&ac[i]);
    }
    scanf("%d",&cntb);
    for(i=0;i<cntb;++i){
        scanf("%d%lf",&be[i],&bc[i]);
    }
    for(i=0;i<cnta;++i){
        for(j=0;j<cntb;++j){
            cc[ae[i]+be[j]]+=ac[i]*bc[j];
        }
    }
    char s[20];
    for(i=0;i<=2000;++i){
        sprintf(s,"%.1f",cc[i]);
        if(strcmp(s,"0.0") && strcmp(s,"-0.0")) ++n;
    }
    printf("%d",n);
    for(i=2000;i>=0;--i){
        sprintf(s,"%.1f",cc[i]);
        if(strcmp(s,"0.0") && strcmp(s,"-0.0"))
            printf(" %d %.1f",i,cc[i]);
    }
    return 0;
}


```
