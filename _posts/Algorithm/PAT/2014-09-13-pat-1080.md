> 快排，模拟

###题意
按照学校指标和学生的排名及申请录取，输出各校录取结果

###细节
- 记录满（超）额学校最后一名的信息，与之排名相同的也可录取
- 输出结果按学号升序

### 代码
```
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#include<vector>
using namespace std;

struct P{
    int w[5];
    int ge,gi,sum,id;
    bool operator < (const P& t)const{
        if(sum!=t.sum) return sum>t.sum;
        if(ge!=t.ge) return ge>t.ge;
        return id<t.id;
    }
}p[40010];
int cnt[110],tot[110];
int n,m,k;
vector<int> ans[110];
int main(){
    freopen("d.in","r",stdin);
    int i,j,w,lst;
    scanf("%d%d%d",&n,&m,&k);
    for(i=0;i<m;++i) scanf("%d",&tot[i]);
    for(i=0;i<n;++i){
        scanf("%d%d",&p[i].ge,&p[i].gi);
        p[i].id=i;
        for(j=0;j<k;++j) scanf("%d",&p[i].w[j]);
        p[i].sum=p[i].ge+p[i].gi;
    }
    sort(p,p+n);
    for(i=0;i<n;++i){
        //printf("%d: %d %d %d,%d %d %d\n",p[i].id,p[i].ge,p[i].gi,p[i].sum,p[i].w[0],p[i].w[1],p[i].w[2]);
        for(j=0;j<k;++j){
            w=p[i].w[j];
            if(cnt[w]<tot[w]){
                ans[w].push_back(i);
                cnt[w]++;
                break;
            }
            lst=ans[w][cnt[w]-1];
            if(p[i].sum==p[lst].sum && p[i].ge==p[lst].ge){
                ans[w].push_back(i);
                cnt[w]++;
                break;
            }
        }
    }
    for(i=0;i<m;++i){
        for(j=0;j<cnt[i];++j) ans[i][j]=p[ans[i][j]].id;
        sort(ans[i].begin(),ans[i].end());
        int f=1;
        for(j=0;j<cnt[i];++j){
            if(f) f=0;
            else printf(" ");
            printf("%d",ans[i][j]);
        }
        printf("\n");
    }
    return 0;
}
```
