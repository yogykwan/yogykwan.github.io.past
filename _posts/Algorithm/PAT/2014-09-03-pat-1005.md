> 模拟

###题意
各位求和，输出和的各位

###细节
- 按和的位数分类

### 代码
```
#include<cstdio>
#include<cstring>
#include<queue>
#include<iostream>
using namespace std;

char p[10][10]={"zero","one","two","three","four","five","six","seven","eight","nine"};
char s[110];

int main(){
    freopen("d.in","r",stdin);
    int ans=0,i;
    scanf("%s",s);
    for(i=0;s[i];++i){
        ans+=s[i]-'0';
    }
    if(ans>99){
        printf("%s %s %s\n",p[ans/100],p[ans/10%10],p[ans%10]);
    }else if(ans>9){
        printf("%s %s\n",p[ans/10],p[ans%10]);
    }else{
        printf("%s\n",p[ans]);
    }
    return 0;
}

```
