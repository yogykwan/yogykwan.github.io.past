> 图论，树

###题意
求根到叶子结点的和为特定值的路径

###细节
- 终点必须为叶子结点
- 输出数组排序

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int son[110][110],cnt[110],a[110],s[110],d[110],p[110],leaf[110];
struct E{
    int a[110];
    int dep;
}e[110];
bool cmp(E a,E b){
    int i,j;
    for(i=j=0;i<a.dep,j<b.dep;++i,++j){
        if(a.a[i]<b.a[j]) return 0;
        if(a.a[i]>b.a[j]) return 1;
    }
    if(i<a.dep) return 1;
    return 0;
}
void gao(int u){
    int i,v;
    for(i=0;i<cnt[u];++i){
        v=son[u][i];
        p[v]=u;
        d[v]=d[u]+1;
        s[v]=s[u]+a[v];
        gao(v);
    }
}
int main(){
    freopen("d.in","r",stdin);
    int n,i,m,j,sum,u,v,tot=0;
    scanf("%d%d%d",&n,&m,&sum);
    for(i=0;i<n;++i) scanf("%d",&a[i]);
    while(m--){
        scanf("%d",&u);
        leaf[u]=1;
        scanf("%d",&cnt[u]);
        for(i=0;i<cnt[u];++i){
            scanf("%d",&v);
            son[u][i]=v;
        }
    }
    s[0]=a[0];
    gao(0);
    for(i=0;i<n;++i){
        if(!leaf[i] && s[i]==sum){
            u=i;
            for(j=d[i]-1;j>=0;--j){
                e[tot].a[j]=a[u];
                u=p[u];
            }
            e[tot++].dep=d[i];
        }
    }
    sort(e,e+tot,cmp);
    for(i=0;i<tot;++i){
        printf("%d",a[0]);
        for(j=0;j<e[i].dep;++j) printf(" %d",e[i].a[j]);
        printf("\n");
    }
    return 0;
}
```
