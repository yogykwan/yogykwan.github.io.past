> 数据结构，集合

###题意
求某两个集合的交集、并集大小

###细节
- 使用set_intersection()函数求交集
- 并集大小由交集直接s计算

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<set>
using namespace std;
set<int> s[51],now;
int main(){
    freopen("d.in","r",stdin);
    int n,m,i,j,a,b;
    scanf("%d",&n);
    for(i=1;i<=n;++i){
        scanf("%d",&m);
        while(m--){
            scanf("%d",&j);
            s[i].insert(j);
        }
    }
    scanf("%d",&m);
    while(m--){
        scanf("%d%d",&i,&j);
        now.clear();
        set_intersection(s[i].begin(),s[i].end(),s[j].begin(),s[j].end(),inserter(now,now.end()));
        a=now.size();
        b=s[i].size()+s[j].size()-a;
        printf("%.1f%%\n",100.0*a/b);
    }
    return 0;
}

```
