> 模拟，进制转换

###题意
求某数某进制表达式是否是回文并输出

###细节
- 注意0

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<queue>
#include<algorithm>
#include<cmath>
#include<string>
using namespace std;
int c[50],cnt;
int main(){
    freopen("d.in","r",stdin);
    int a,b,i,ok;
    scanf("%d%d",&a,&b);
    do{
        c[cnt++]=a%b;
        a/=b;
    }while(a);
    for(i=0;i<cnt/2+1;++i){
        if(c[i]!=c[cnt-1-i]) break;
    }
    printf("%s\n",i==cnt/2+1?"Yes":"No");
    for(i=cnt-1;i;--i) printf("%d ",c[i]);
    printf("%d\n",c[0]);
    return 0;
}
```
