> 数学，哈希

###题意
对应给出哈希值，使用正向的二次探测解决碰撞

###细节
- 正向二次探测即将[0,mod)的平方依次加入尝试
- 双向二次探测，先加后减

### 代码
```
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#include<queue>
using namespace std;

int vis[10020];

bool ip(int a){
    if(a<2) return 0;
    for(int i=2;i*i<=a;++i)
        if(a%i==0) return 0;
    return 1;
}
int main(){
    freopen("d.in","r",stdin);
    int n,m,i,j,ans;
    scanf("%d%d",&m,&n);
    while(!ip(m)) ++m;
    scanf("%d",&i);
    i%=m;
    for(j=0;j<m;++j){
        ans=(i+j*j)%m;
        if(!vis[ans]){
            vis[ans]=1;
            printf("%d",ans);
            break;
        }
    }
    while(--n){
        scanf("%d",&i);
        i%=m;
        for(j=0;j<m;++j){
            ans=(i+j*j)%m;
            if(!vis[ans]){
                vis[ans]=1;
                printf(" %d",ans);
                break;
            }
        }
        if(j==m) printf(" -");
    }
    return 0;
}

```
