> 模拟，大数

###题意
反复将数与它的反文相加，直至回文出现

###细节
- 必须用大数

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<queue>
#include<algorithm>
#include<map>
#include<string>
#include <iomanip>
#include<set>
using namespace std;

int a[61],b[61];
char s[61];
bool ok(){
    int i;
    for(i=0;i<=60;++i) if(a[i]!=b[i]) return 0;
    return 1;
}
void gao(){
    int i,j;
    for(i=60;;--i) if(a[i]) break;
    for(j=0;j<=i;++j) b[j]=a[i-j];
}
void add(){
    int i,j=0;
    for(i=0;i<=60;++i){
        a[i]=a[i]+b[i]+j;
        j=a[i]/10;
        a[i]%=10;
    }
}
void out(){
    int i;
    for(i=60;;--i) if(a[i]) break;
    for(;i>=0;--i) printf("%d",a[i]);
    printf("\n");
}
int main(){
    freopen("d.in","r",stdin);
    int i,n,len;
    scanf("%s%d",s,&n);
    len=strlen(s);
    for(i=0;i<len;++i){
        a[i]=s[len-1-i]-'0';
    }
    gao();
    if(ok()){
        printf("%s\n%d\n",s,0);
        return 0;
    }
    for(i=1;i<=n;++i){
        add();
        gao();
        if(ok()) break;
    }
    if(i<=n){
        out();
        printf("%d\n",i);
    }else{
        out();
        printf("%d\n",n);
    }
    return 0;
}

```
