> 数据结构

###题意
求每人办完业务的时刻

###细节
- 17点前开始办理的必须办完
- 每分钟模拟正在办理业务的人剩余时间

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<queue>
#include<iostream>
using namespace std;

int n,m,tot,k;
queue<int> q[25];
int t[25];
int ans[1010];
int nd[1010];

int main(){
    freopen("d.in","r",stdin);
    int i,j,now=0,cnt=0,ok=0;
    scanf("%d%d%d%d",&n,&m,&tot,&k);
    for(i=0;i<tot;++i) scanf("%d",&nd[i]);
    for(i=0;i<m;++i){
        for(j=0;j<n;++j){
            if(cnt<tot) q[j].push(cnt++);
            else goto L;
        }
    }
    L:;
    for(i=0;i<n;++i) if(!q[i].empty()) t[i]=nd[q[i].front()];
    for(now=1;now<540;++now){
        for(i=0;i<n;++i){
            if(--t[i]==0){
                ans[q[i].front()]=now;
                q[i].pop();
                if(cnt<tot) q[i].push(cnt++);
                if(!q[i].empty()) t[i]=nd[q[i].front()];
                if(++ok==tot) goto LL;
            }
        }
    }
    for(i=0;i<n;++i){
        if(!q[i].empty()){
            ans[q[i].front()]=now=539+t[i];
        }
    }
    LL:;
    while(k--){
        scanf("%d",&i);
        --i;
        if(ans[i]==0) printf("Sorry\n");
        else printf("%02d:%02d\n",8+ans[i]/60,ans[i]%60);
    }
    return 0;
}


```
