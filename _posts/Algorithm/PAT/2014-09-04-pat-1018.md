> 图论

###题意
求最短路径、送车数、收车数

###细节
- 只有去程可以调整车数
- 所选路径上所有点都需调整
- 求出所有最短路后，递归更新，同一节点可能递归多次
- 在送车数相同时，选收车数少的方案

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<queue>
#include<algorithm>
#include<cmath>
#include<string>
using namespace std;

const int inf=0x7fffffff;
int d[510],h[510],head[510],vis[510];
int s=inf,b,tot,l;
int q[510][510];
int son[510];
int path[510],ans[510];
struct E{
    int v,w,next;
}e[300000];
int cnt;
void add(int u,int v,int w){
    e[cnt].next=head[u];
    e[cnt].v=v;
    e[cnt].w=w;
    head[u]=cnt++;
}
typedef pair<int,int> pii;
priority_queue<pii,vector<pii>,greater<pii> > pq;
void gao(int n){
    int i,ts=0,tb=0,u;
    //printf("0");for(i=n-2;i>=0;--i) printf("->%d",path[i]);printf("\n");
    for(i=n-2;i>=0;--i){
        u=h[path[i]]-l;
        if(u>0) tb+=u;
        else if(u<0){
            if(u+tb>=0) tb+=u;
            else{
                ts-=u+tb;
                tb=0;
            }
        }
    }
    if(ts<s || ts==s && tb<b){
        s=ts;b=tb;tot=n-1;
        for(i=0;i<n;++i) ans[i]=path[i];
    }
}
void dfs(int v,int deep){
    if(v==0){
        gao(deep);
        return;
    }
    int i,u;
    for(i=0;i<son[v];++i){
        u=q[v][i];
        path[deep]=u;
        dfs(u,deep+1);
    }
}

int main(){
    freopen("d.in","r",stdin);
    int i,j,t,n,m,u,v,w,ed,now,temp;
    memset(head,-1,sizeof(head));
    scanf("%d%d%d%d",&l,&n,&t,&m);
    l>>=1;
    for(i=1;i<=n;++i) scanf("%d",&h[i]);
    for(i=0;i<m;++i){
        scanf("%d%d%d",&u,&v,&w);
        add(u,v,w);
        add(v,u,w);
    }
    for(i=0;i<=n;++i) d[i]=inf;
    d[0]=0;
    pq.push(make_pair(0,0));
    while(!pq.empty()){
        u=pq.top().second;
        pq.pop();
        if(vis[u]) continue;
        vis[u]=1;
        if(u==t) break;
        for(ed=head[u];ed!=-1;ed=e[ed].next){
            v=e[ed].v;
            if(vis[v]) continue;
            w=e[ed].w;
            if(d[u]+w<d[v]){
                son[v]=1;
                q[v][0]=u;
                d[v]=d[u]+w;
                pq.push(make_pair(d[v],v));
            }else if(d[u]+w==d[v]){
                q[v][son[v]++]=u;
            }
        }
    }
    path[0]=t;
    dfs(t,1);
    printf("%d 0",s);
    for(--tot;tot>=0;--tot) printf("->%d",ans[tot]);
    printf(" %d\n",b);
    return 0;
}
```
