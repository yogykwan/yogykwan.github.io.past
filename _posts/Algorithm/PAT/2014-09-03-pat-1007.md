> 动态规划

###题意
求最大子串和及首尾元素

###细节
- 用数列第一个数初始化最大和

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
int s[10010],h[10010],a[10010];
int main(){
    freopen("d.in","r",stdin);
    int n,i;
    scanf("%d",&n);
    for(i=0;i<n;++i) scanf("%d",a+i);
    s[0]=a[0];
    int mm=s[0];
    int id=0;
    for(i=1;i<n;++i){
        if(s[i-1]>=0){
            h[i]=h[i-1];
            s[i]=s[i-1]+a[i];
        }else{
            h[i]=i;
            s[i]=a[i];
        }
        if(s[i]>mm || s[i]==mm && h[i]<h[id]){
            mm=s[i];
            id=i;
        }
    }
    if(mm<0) printf("0 %d %d\n",a[0],a[n-1]);
    else printf("%d %d %d\n",mm,a[h[id]],a[id]);
    return 0;
}


```
