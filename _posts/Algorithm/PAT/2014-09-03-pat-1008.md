> 模拟

###题意
求总时间

###细节
- 同楼多次停留算多次

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

int main(){
    freopen("d.in","r",stdin);
    int n,i,ans,j=0;
    scanf("%d",&n);
    ans=n*5;
    while(n--){
        scanf("%d",&i);
        if(i>j) ans+=6*(i-j);
        else if(i<j) ans+=4*(j-i);
        j=i;
    }
    printf("%d\n",ans);
    return 0;
}


```
