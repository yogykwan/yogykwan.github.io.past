> 快排，贪心

###题意
给出各种月饼的库存和总价，求最大销售额

###细节
- 市场需求量与库存总量的关系任意
- 按单价快排

### 代码
```
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
struct P{
    double store,price;
    bool operator <(const P&t)const{
        return price*t.store>t.price*store;
    }
}p[1010];
int main(){
    freopen("d.in","r",stdin);
    int n,i;
    double m,ans=0;
    scanf("%d%lf",&n,&m);
    for(i=0;i<n;++i) scanf("%lf",&p[i].store);
    for(i=0;i<n;++i) scanf("%lf",&p[i].price);
    sort(p,p+n);
    i=-1;
    while(m>0 && ++i<n){
        if(p[i].store>=m){
            ans+=p[i].price*m/p[i].store;
            break;
        }
        ans+=p[i].price;
        m-=p[i].store;
    }
    printf("%.2f",ans);
    return 0;
}

```
