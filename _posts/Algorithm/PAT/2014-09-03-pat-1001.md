## AC代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int s[20];
int main(){
    freopen("d.in","r",stdin);
    int a,b,c,i=0,d;
    scanf("%d%d",&a,&b);
    c=a+b;
    if(c<1000 && c>-1000){
        printf("%d\n",c);
    }else{
        if(c<0){
            printf("-");
            c=-c;
        }
        while(c){
            s[i++]=c%10;
            c/=10;
        }

        d=i%3;
        if(!d) d=3;
        while(d--){
            printf("%d",s[--i]);
        }
        while(i){
            d=3;
            printf(",");
            while(d--){
                printf("%d",s[--i]);
            }
        }
    }
    return 0;
}
```
