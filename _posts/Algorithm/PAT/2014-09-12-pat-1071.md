> 模拟，字符串

###题意
统计出现频率最高的单词

###细节
- 字母大写转小写，非字母为分隔符
- 用map映射string到int

### 代码
```
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#include<map>
using namespace std;
char s[1048600];
map<string,int> mymap;
string name;
int main(){
    freopen("d.in","r",stdin);
    gets(s);
    int b=0,i,t,ans=0,temp;
    for(i=0;s[i];++i){
        b=0;
        while(s[i]){
            if(isalpha(s[i]) || isdigit(s[i])) break;
            ++i;
        }
        b=i;
        while(s[i]){
            if(isalpha(s[i])) s[i]=tolower(s[i]);
            else if(!isdigit(s[i])) break;
            ++i;
        }
        string word(s+b,s+i);
        temp=++mymap[word];
        if(temp>ans || temp==ans && word<name){
            name=word;
            ans=temp;
        }
    }
    cout<<name<<" "<<ans;
    return 0;
}

```
