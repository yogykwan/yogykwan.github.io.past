> 数据结构，二叉树

###题意
求给出序列组成的唯一的二叉搜索树

###细节
- 按层构造树，结点数控制大小
- 中序遍历将排序后的键值加入树

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int two[16],l[1010],r[1010],a[1010],p[1010],cnt;
void gao(int u){
    if(l[u]!=-1) gao(l[u]);
    p[u]=a[cnt++];
    if(r[u]!=-1) gao(r[u]);
}
int main(){
    freopen("d.in","r",stdin);
    int n,i,m,j,d;
    two[0]=1;
    for(i=1;i<=15;++i) two[i]=two[i-1]<<1;
    memset(l,-1,sizeof(l));
    memset(r,-1,sizeof(r));
    scanf("%d",&n);
    for(i=0;i<n;++i) scanf("%d",a+i);
    sort(a,a+n);
    i=1;
    d=0;
    while(i<n){
        for(j=two[d]-1;j<=two[d+1]-2 && i<n;++j){
            l[j]=i++;
            if(i<n) r[j]=i++;
        }
        ++d;
    }
    gao(0);
    printf("%d",p[0]);
    for(i=1;i<n;++i) printf(" %d",p[i]);
    return 0;
}

```
