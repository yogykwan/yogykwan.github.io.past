> 图论

###题意
求最短路条数、携带救援队数

###细节
- 将条数、队数与距离一起DP
- 距离增大刷新条数
- 距离相同增大条数，进而队数增大更新队数

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;

typedef pair<int,int> pii;
priority_queue<pii,vector<pii>,greater<pii> > pq;
const int inf=0x7fffffff;
int vis[510];
int d[510];
int p[510];
struct E{
    int v,w;
    int next;
}e[1000000];
int h[510];
int cnt;
int a[510],r[510];
void add(int u,int v,int w){
    e[cnt].next=h[u];
    e[cnt].v=v;
    e[cnt].w=w;
    h[u]=cnt++;
}

int main(){
    freopen("d.in","r",stdin);
    int n,m,s,t,u,v,w,i;
    scanf("%d%d%d%d",&n,&m,&s,&t);
    for(i=0;i<n;++i){
        scanf("%d",&a[i]);
        r[i]=a[i];
    }
    memset(h,-1,sizeof(h));
    for(i=0;i<m;++i){
        scanf("%d%d%d",&u,&v,&w);
        add(u,v,w);
        add(v,u,w);
    }
    for(i=0;i<n;++i) d[i]=inf;
    d[s]=0;
    pq.push(make_pair(0,s));
    pii temp;
    p[s]=1;
    while(!pq.empty()){
        u=pq.top().second;
        pq.pop();
        if(u==t) break;
        if(vis[u]) continue;
        vis[u]=1;
        int ed;
        for(ed=h[u];ed!=-1;ed=e[ed].next){
            v=e[ed].v;
            if(vis[v]) continue;
            w=e[ed].w;
            if(d[u]+w<d[v]){
                p[v]=p[u];
                d[v]=d[u]+w;
                r[v]=r[u]+a[v];
                pq.push(make_pair(d[v],v));
            }else if(d[u]+w==d[v]){
                p[v]+=p[u];
                if(r[u]+w>r[v]){
                    d[v]=d[u]+w;
                    r[v]=r[u]+a[v];
                    pq.push(make_pair(d[v],v));
                }
            }
        }
    }
    printf("%d %d\n",p[t],r[t]);
    return 0;
}
```
