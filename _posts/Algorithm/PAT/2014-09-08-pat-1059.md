> 数学，分解质因数

###题意
给出一个数，对它分解质因数

###细节
- 打表10万内素数
- 注意1

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
#include<stack>
using namespace std;
bool np[100010];
int p[100010],cnt;
void pp(){
    int i,j;
    long long t;
    for(i=2;i<100000;++i){
        if(!np[i]) p[cnt++]=i;
        for(j=0;j<cnt && (t=1LL&p[j]*i)<100000;++j){
            np[t]=1;
            if(i%p[j]) break;
        }
    }
}
int main(){
    freopen("d.in","r",stdin);
    int i,n,b,f=1;
    pp();
    scanf("%d",&n);
    if(n==1){
        printf("1=1");
        return 0;
    }
    printf("%d=",n);
    for(i=0;i<cnt && n>1;++i){
        if(n%p[i]) continue;
        b=1;
        n/=p[i];
        while(n%p[i]==0){
            ++b;
            n/=p[i];
        }
        if(f) f=0;
        else printf("*");
        printf("%d",p[i]);
        if(b>1) printf("^%d",b);
    }
    if(n>1){
        if(f) f=0;
        else printf("*");
        printf("%d",n);
    }
    return 0;
}

```
