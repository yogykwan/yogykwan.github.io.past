###题意
含三个单位的钱数求和输出

###细节
- 进制不同

#include<cstdio>
#include<cstring>

    freopen("d.in","r",stdin);
    int j;
    scanf("%d.%d.%d %d.%d.%d",&a[0],&b[0],&c[0],&a[1],&b[1],&c[1]);
    c[2]=c[0]+c[1];
    j=c[2]/29;
    c[2]%=29;

}
