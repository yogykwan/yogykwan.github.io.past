> 图论，最短路

###题意
按最短路程、最大欢乐值、最大平均欢乐值的优先级求最优路径

###细节
- 更新最短路条数时累加所有最短前趋的路径数

### 代码
```
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#include<queue>
#include<map>
using namespace std;
int s[210],a[210],d[210],tot[210],len[210];
const int inf=0x7fffffff;
typedef pair<int,int> pii;
priority_queue<pii,vector<pii>,greater<pii> > pq;
string name[210];
int ans[210];
map<string,int> mymap;
struct E{
    int v,w,nxt;
}e[40010];
int head[210],vis[210],pre[210];
string aa,bb;
int cnt;
void add(int u,int v,int w){
    e[cnt].v=v;
    e[cnt].w=w;
    e[cnt].nxt=head[u];
    head[u]=cnt++;
}
int main(){
    freopen("d.in","r",stdin);
    int n,i,u,v,w,t=1,j,m;
    cin>>n>>m>>name[1];
    mymap[name[1]]=1;
    for(i=2;i<=n;++i){
        cin>>name[i]>>a[i];
        if(name[i]=="ROM") t=i;
        mymap[name[i]]=i;
    }
    memset(head,-1,sizeof(head));
    memset(pre,-1,sizeof(pre));
    for(i=1;i<=n;++i) d[i]=inf;
    while(m--){
        cin>>aa>>bb>>w;
        u=mymap[aa];
        v=mymap[bb];
        add(u,v,w);
        add(v,u,w);
    }
    d[1]=0;
    tot[1]=1;
    pq.push(make_pair(0,1));
    while(!pq.empty()){
        u=pq.top().second;
        pq.pop();
        if(u==t) break;
        if(vis[u]) continue;
        vis[u]=1;
        for(i=head[u];i!=-1;i=e[i].nxt){
            v=e[i].v;
            w=e[i].w;
            if(vis[v]) continue;
            if(d[u]+w<d[v]){
                d[v]=d[u]+w;
                s[v]=s[u]+a[v];
                tot[v]=tot[u];
                len[v]=len[u]+1;
                pq.push(make_pair(d[v],v));
                pre[v]=u;
            }
            else if(d[u]+w==d[v]){
                tot[v]+=tot[u];
                if(s[u]+a[v]>s[v] || s[u]+a[v]==s[v] && len[u]+1<len[v]){
                    s[v]=s[u]+a[v];
                    len[v]=len[u]+1;
                    pq.push(make_pair(d[v],v));
                    pre[v]=u;
                }
            }
        }
    }
    printf("%d %d %d %d\n",tot[t],d[t],s[t],s[t]/len[t]);
    j=0;
    i=pre[t];
    while(i!=-1){
        ans[j++]=i;
        i=pre[i];
    }
    for(i=j-1;i>=0;--i){
        cout<<name[ans[i]]<<"->";
    }
    cout<<"ROM";
    return 0;
}
```
