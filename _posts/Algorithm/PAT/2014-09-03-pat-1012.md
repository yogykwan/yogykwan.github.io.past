> 排序

### 题意
求每个人最优排名及科目

### 细节
- 写4个不同的快排函数

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cstdlib>
using namespace std;

int a[1000000],c[1000000],m[1000000],e[1000000];

struct P{
    int id,c,m,e,a;
}p[2010];
bool cmpa(P a,P b){ return a.a>b.a; }
bool cmpc(P a,P b){ return a.c>b.c; }
bool cmpm(P a,P b){ return a.m>b.m; }
bool cmpe(P a,P b){ return a.e>b.e; }

int main(){
    freopen("d.in","r",stdin);
    int n,t,r,i;
    char o;
    scanf("%d%d",&n,&t);
    for(i=0;i<n;++i){
        scanf("%d%d%d%d",&p[i].id,&p[i].c,&p[i].m,&p[i].e);
        p[i].a=p[i].c+p[i].m+p[i].e;
    }
    sort(p,p+n,cmpa);
    a[p[0].id]=1;
    for(i=1;i<n;++i){
        if(p[i].a==p[i-1].a) a[p[i].id]=a[p[i-1].id];
        else a[p[i].id]=i+1;
    }
    sort(p,p+n,cmpc);
    c[p[0].id]=1;
    for(i=1;i<n;++i){
        if(p[i].c==p[i-1].c) c[p[i].id]=c[p[i-1].id];
        else c[p[i].id]=i+1;
    }
    sort(p,p+n,cmpm);
    m[p[0].id]=1;
    for(i=1;i<n;++i){
        if(p[i].m==p[i-1].m) m[p[i].id]=m[p[i-1].id];
        else m[p[i].id]=i+1;
    }
    sort(p,p+n,cmpe);
    e[p[0].id]=1;
    for(i=1;i<n;++i){
        if(p[i].e==p[i-1].e) e[p[i].id]=e[p[i-1].id];
        else e[p[i].id]=i+1;
    }
    while(t--){
        scanf("%d",&i);
        if(a[i]==0){
            printf("N/A\n");
            continue ;
        }
        r=a[i];o='A';
        if(c[i]<r){
            r=c[i];o='C';
        }
        if(m[i]<r){
            r=m[i];o='M';
        }
        if(e[i]<r){
            r=e[i];o='E';
        }
        printf("%d %c\n",r,o);
    }
    return 0;
}


```
