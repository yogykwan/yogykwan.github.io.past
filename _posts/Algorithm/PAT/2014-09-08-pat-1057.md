> 数据结构，树状数组

###题意
给出栈的相关操作，在线询问中位数

###细节
- 树状数组求第k大数
- k不满足栈内元素数量直接否定

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
#include<stack>
using namespace std;
stack<int> ms;
int s[100010],n;
inline int lowbit(int x){
    return x&-x;
}
void update(int x,int c){
    while(x<=n){
        s[x]+=c;
        x+=lowbit(x);
    }
}
int getsum(int x){
    int ans=0;
    while(x){
        ans+=s[x];
        x-=lowbit(x);
    }
    return ans;
}
int getk(int k){
    int ans=0,i;
    for(i=20;i>=0;--i){
        ans|=1<<i;
        if(ans>=n || s[ans]>=k) ans^=1<<i;
        else k-=s[ans];
    }
    return ans+1;
}
char op[12];
int main(){
    freopen("d.in","r",stdin);
    n=100000;
    int i,m,cnt=0;
    scanf("%d",&m);
    while(m--){
        scanf("%s",op);
        if(op[1]=='u'){
            scanf("%d",&i);
            ms.push(i);
            update(i,1);
            ++cnt;
        }else if(op[1]=='o'){
            if(!cnt) printf("Invalid\n");
            else{
                update(ms.top(),-1);
                printf("%d\n",ms.top());
                ms.pop();
                --cnt;
            }
        }else{
            if(!cnt) printf("Invalid\n");
            else{
                if(cnt&1) i=(cnt+1)/2;
                else i=cnt/2;
                printf("%d\n",getk(i));
            }
        }
    }
    return 0;
}
```
