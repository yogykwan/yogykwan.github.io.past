> 模拟，大数

###题意
给出最多9位的整数，输出中文读法

###细节
- 连续的0只读一次，且其后必须包含非零元素
- 后八位分成两部分操作

### 代码
```
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#include<vector>
#include<cmath>
using namespace std;
char s[15],n[15];
char word[10][10]={"ling","yi","er","san","si","wu","liu","qi","ba","jiu"};
char danwei[10][10]={"Qian","Bai","Shi"};
int k,j;
char now[5];
void out(char *s){
    if(!k) k=1;
    else printf(" ");
    printf("%s",s);
}
void gao(char *a,int b){
    int i,flag=0,ten=1000;
    for(i=0;i<4;++i){
        if(a[i]>'0') break;
        ten/=10;
    }
    if(i==4) return;
    for(;i<4;++i){
        if(a[i]=='0'){
            if(!flag){
                flag=1;
                out("ling");
            }
        }else{
            out(word[a[i]-48]);
            if(i<3) out(danwei[i]);
        }
        if(b%ten==0) return;
        ten/=10;
    }
}
int main(){
    freopen("d.in","r",stdin);
    int i,b=0,a;
    scanf("%s",s);
    sscanf(s,"%d",&a);
    if(s[0]=='-'){
        out("Fu");
        b=1;
        a=-a;
    }
    if(a==0){
        out("ling");
        return 0;
    }
    a%=100000000;
    for(i=strlen(s)-1;i>=b;--i) n[j++]=s[i];
    for(i=8;;--i){
        if(n[i]!='0') break;
        else n[i]=0;
    }
    if(n[8]>'0'){
        out(word[n[8]-48]);
        out("Yi");
        if(n[7]=='0' && a) out("ling");
    }
    now[0]=n[7],now[1]=n[6],now[2]=n[5],now[3]=n[4];
    gao(now,a/10000);
    if(a/10000){
        out("Wan");
        if(n[3]=='0' && a%10000) out("ling");
    }
    now[0]=n[3],now[1]=n[2],now[2]=n[1],now[3]=n[0];
    gao(now,a%10000);
    return 0;
}

```
