> 排序，快排

###题意
求每人的单总排名、场次、场排名

###细节
- 注意并列情况

### 代码
```
#include<cstdio>
#include<cstring>
#include<iostream>
#include<queue>
#include<algorithm>
#include<map>
#include<string>
#include <iomanip>
#include<set>
using namespace std;
typedef long long ll;
struct E{
    ll name;
    int s;
}e[310];
bool cmp(E a,E b){
    if(a.s!=b.s) return a.s>b.s;
    return a.name<b.name;
}
struct P{
    ll name;
    int s,r,l;
}p[30010];
bool cmpp(P a,P b){
    if(a.s!=b.s) return a.s>b.s;
    return a.name<b.name;
}
int main(){
    freopen("d.in","r",stdin);
    int n,i,j,m,rk,N,cnt=0;
    scanf("%d",&n);
    for(i=1;i<=n;++i){
        scanf("%d",&m);
        for(j=0;j<m;++j) scanf("%lld%d",&e[j].name,&e[j].s);
        sort(e,e+m,cmp);
        for(j=0;j<m;++j){
            if(!j || e[j].s!=e[j-1].s) rk=j+1;
            p[cnt].name=e[j].name;
            p[cnt].s=e[j].s;
            p[cnt].r=rk;
            p[cnt++].l=i;
        }
    }
    N=cnt;
    sort(p,p+N,cmpp);
    printf("%d\n",N);
    for(i=0;i<N;++i){
        if(!i || p[i].s!=p[i-1].s) rk=i+1;
        printf("%013lld %d %d %d\n",p[i].name,rk,p[i].l,p[i].r);
    }
    return 0;
}
```
